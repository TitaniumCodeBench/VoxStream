useEffect(() => {
  const ws = new WebSocket("ws://localhost:8000/ws");

  ws.onmessage = (event) => {
    setText((prev) => prev + " " + event.data);
  };

  ws.onerror = (err) => console.error("WS error:", err);
  ws.onclose = () => console.log("WS closed");

  return () => ws.close();
}, []);

useEffect(() => {
  const ws = new WebSocket("ws://localhost:8000/st");

  ws.onmessage = (event) => {
    const data_ = JSON.parse(event.data);
    if (data_ && data_.message) {
      setIs_Recording(data_.message.is_recording);
      setIs_Running(data_.message.is_running);
      setIs_Shut_Down(data_.message.is_shut_down);
    }
  };

  ws.onerror = (err) => console.error("WS error:", err);
  ws.onclose = () => console.log("WS closed");

  return () => ws.close();
}, []);
